   ```astro
     <div class="filter-pills mb-4 flex flex-wrap gap-2">

       ))}

   )}
   ```

3. Add client-side filtering JavaScript before closing </Layout>:
   ```astro
   <script>
     document.addEventListener('DOMContentLoaded', () => {
       const pills = document.querySelectorAll('.filter-pill');
       const entries = document.querySelectorAll('.entry-list li');
       
       pills.forEach(pill => {
         pill.addEventListener('click', () => {
           const tag = pill.dataset.tag;
           const isActive = pill.classList.contains('active');
           
           // Toggle active state
           if (tag === 'all') {
             pills.forEach(p => p.classList.remove('active'));
             pill.classList.add('active');
           } else {
             document.querySelector('[data-tag="all"]').classList.remove('active');
             pill.classList.toggle('active');
           }
           
           // Get selected tags
           const selected = [...document.querySelectorAll('.filter-pill.active')]
             .map(p => p.dataset.tag)
             .filter(t => t !== 'all');
           
           // Filter entries
           entries.forEach(entry => {
             const entryTags = entry.dataset.tags?.split(',') || [];
             if (selected.length === 0 || selected.some(t => entryTags.includes(t))) {
               entry.style.display = '';
             } else {
               entry.style.display = 'none';
             }
           });
         });
       });
     });
   </script>
   ```

4. Add data-tags attribute to each entry <li>:
   ```astro
   <li data-tags={entry.data.tags?.join(',')}>
   ```

5. Add CSS for filter pills in global.css (or style block):
